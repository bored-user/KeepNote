const multer = require('multer');
module.exports.upload = multer({
  • storage: multer.memoryStorage() // keep uploaded files temporarily stored on memory
  • fileFilter = (req, file, next) => {

In said error object, one can specify a message to be displayed later - next({message: 'message'}, false).

  • }
