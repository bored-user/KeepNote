Written between square brackets. Are able to store multiple data types, even other lists.


myList = [1,"hi",variable]

print(myList[0]) # 1
print(myList[0:2]) # Will print items indexed by 0 and 1 (2 NOT INCLUDED)
print(myList[-1]) # Print last item
print(myList[1:]) # Prints item indexed by 1 until the last indexed item

print(len(myList)) # 3

myList.append("hello") # "hello" will be added to the list, as the last item

myList.pop() # Removes the last item
myList.pop(0) # Removes the first item

mySecondList = ["pair item1", "pair item2", " pair item3"]
myFinalList = zip(myList, mySecondList)
print(list(myFinalList)) # [(1, "pair item 1"), ("hi", "pair item 2"), (variable, "pair item 3")]
                          • # If one list has more items than the other, the items in excess are simply ignored (it will only display pair of values).
